# Aegis — OWS Policy, Commerce & Visibility — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Aegis — the first ecosystem extension for OWS — delivering policy executables, a commerce MCP server, a CLI, and a real-time Sentinel Dashboard.

**Architecture:** Turborepo monorepo with 5 packages: `shared` (types + utils), `policies` (3 executables), `mcp-server` (6 MCP tools), `cli` (Commander.js), and `dashboard` (Next.js 15). All data stored as JSON files in `~/.ows/aegis/`. Policies are standalone Node.js scripts invoked by OWS's policy engine via stdin/stdout JSON. The MCP server uses `@modelcontextprotocol/sdk` with stdio transport. The dashboard reads `~/.ows/aegis/` via API routes.

**Tech Stack:** TypeScript 5.x, Node.js 20+, Turborepo, `@open-wallet-standard/core`, `@modelcontextprotocol/sdk`, Zod 3, Commander.js, Next.js 15, Tailwind CSS 4, shadcn/ui, Recharts.

**Spec reference:** `AEGIS_OWS_HACKATHON_BUILD_SPEC.md` in repo root.

import { readFileSync, writeFileSync } from "node:fs";
import { existsSync } from "node:fs";
import { PATHS, ensureAegisDir } from "./paths.js";
import type { BudgetLedger, LedgerEntry } from "./types.js";

export function readLedger(): BudgetLedger {
  if (!existsSync(PATHS.budgetLedger)) {
    return { entries: [] };
  }
  const raw = readFileSync(PATHS.budgetLedger, "utf-8");
  return JSON.parse(raw) as BudgetLedger;
}

export function appendLedgerEntry(entry: LedgerEntry): void {
  ensureAegisDir();
  const ledger = readLedger();
  ledger.entries.push(entry);
  writeFileSync(PATHS.budgetLedger, JSON.stringify(ledger, null, 2));
}

export function getSpentInPeriod(
  ledger: BudgetLedger,
  apiKeyId: string,
  chainId: string,
  token: string,
  period: "daily" | "weekly" | "monthly"
): number {
  const now = new Date();
  let cutoff: Date;

  switch (period) {
    case "daily":
      cutoff = new Date(now.getFullYear(), now.getMonth(), now.getDate());
      break;
    case "weekly": {
      const day = now.getDay();
      cutoff = new Date(now.getFullYear(), now.getMonth(), now.getDate() - day);
      break;
    }
    case "monthly":
      cutoff = new Date(now.getFullYear(), now.getMonth(), 1);
      break;
  }

  return ledger.entries
    .filter((e) => {
      if (new Date(e.timestamp) < cutoff) return false;
      if (apiKeyId !== "*" && e.apiKeyId !== apiKeyId) return false;
      if (chainId !== "*" && e.chainId !== chainId) return false;
      if (token !== "*" && e.token !== token) return false;
      return true;
    })
    .reduce((sum, e) => sum + parseFloat(e.amount), 0);
}

// packages/shared/src/approvals.ts
import { readFileSync, writeFileSync, existsSync } from "node:fs";
import { randomUUID } from "node:crypto";
import { PATHS, ensureAegisDir } from "./paths.js";
import type { ApprovalQueue, PendingApproval, SerializedTransaction } from "./types.js";

export function readApprovals(): ApprovalQueue {
  if (!existsSync(PATHS.pendingApprovals)) {
    return { pending: [] };
  }
  const raw = readFileSync(PATHS.pendingApprovals, "utf-8");
  return JSON.parse(raw) as ApprovalQueue;
}

function writeApprovals(queue: ApprovalQueue): void {
  ensureAegisDir();
  writeFileSync(PATHS.pendingApprovals, JSON.stringify(queue, null, 2));
}

export function addPendingApproval(params: {
  apiKeyId: string;
  chainId: string;
  transaction: SerializedTransaction;
  estimatedValue: string;
  token: string;
  reason?: string;
  ttlMinutes: number;
}): PendingApproval {
  const now = new Date();
  const approval: PendingApproval = {
    id: `tx-${randomUUID().slice(0, 8)}`,
    apiKeyId: params.apiKeyId,
    chainId: params.chainId,
    transaction: params.transaction,
    estimatedValue: params.estimatedValue,
    token: params.token,
    reason: params.reason,
    createdAt: now.toISOString(),
    expiresAt: new Date(now.getTime() + params.ttlMinutes * 60_000).toISOString(),
    status: "pending",
  };

  const queue = readApprovals();
  queue.pending.push(approval);
  writeApprovals(queue);
  return approval;
}

export function resolveApproval(id: string, status: "approved" | "rejected"): PendingApproval | null {
  const queue = readApprovals();
  const item = queue.pending.find((a) => a.id === id);
  if (!item || item.status !== "pending") return null;

  item.status = status;
  item.resolvedAt = new Date().toISOString();
  writeApprovals(queue);
  return item;
}

export function isApproved(apiKeyId: string, chainId: string, txTo?: string): boolean {
  const queue = readApprovals();
  return queue.pending.some(
    (a) =>
      a.status === "approved" &&
      a.apiKeyId === apiKeyId &&
      a.chainId === chainId &&
      a.transaction.to === txTo &&
      new Date(a.expiresAt) > new Date()
  );
}

// packages/shared/src/registry.ts
import { readFileSync, writeFileSync, existsSync } from "node:fs";
import { randomUUID } from "node:crypto";
import { PATHS, ensureAegisDir } from "./paths.js";
import type { ServiceRegistry, ServiceEntry } from "./types.js";

export function readRegistry(): ServiceRegistry {
  if (!existsSync(PATHS.serviceRegistry)) {
    return { services: [] };
  }
  const raw = readFileSync(PATHS.serviceRegistry, "utf-8");
  return JSON.parse(raw) as ServiceRegistry;
}

function writeRegistry(registry: ServiceRegistry): void {
  ensureAegisDir();
  writeFileSync(PATHS.serviceRegistry, JSON.stringify(registry, null, 2));
}

export function registerService(params: Omit<ServiceEntry, "id" | "registeredAt">): ServiceEntry {
  const entry: ServiceEntry = {
    ...params,
    id: `svc-${randomUUID().slice(0, 8)}`,
    registeredAt: new Date().toISOString(),
  };
  const registry = readRegistry();
  registry.services.push(entry);
  writeRegistry(registry);
  return entry;
}

export function searchServices(query: string, maxPrice?: string, chain?: string, protocol?: string): ServiceEntry[] {
  const registry = readRegistry();
  const q = query.toLowerCase();
  return registry.services.filter((s) => {
    if (!s.name.toLowerCase().includes(q) && !s.description.toLowerCase().includes(q)) return false;
    if (maxPrice && parseFloat(s.price) > parseFloat(maxPrice)) return false;
    if (chain && !s.chains.includes(chain)) return false;
    if (protocol && protocol !== "any" && s.protocol !== protocol) return false;
    return true;
  });
}

This is the core spending limits policy. It reads PolicyContext from stdin, checks the budget ledger, and returns PolicyResult to stdout.

```typescript
// packages/policies/src/budget/index.ts
#!/usr/bin/env node

import {
  type PolicyContext,
  type PolicyResult,
  readBudgetConfig,
  readLedger,
  getSpentInPeriod,
  appendPolicyLog,
} from "@aegis-ows/shared";

async function main(): Promise<void> {
  // Read PolicyContext from stdin
  const chunks: Buffer[] = [];
  for await (const chunk of process.stdin) {
    chunks.push(chunk as Buffer);
  }
  const input = Buffer.concat(chunks).toString("utf-8");
  const ctx: PolicyContext = JSON.parse(input);

  const config = readBudgetConfig();
  const ledger = readLedger();

  // Estimate transaction value from simulation or tx data
  const txValue = estimateValue(ctx);

  // Check each matching limit
  for (const limit of config.limits) {
    const chainMatch = limit.chainId === "*" || limit.chainId === ctx.chainId;
    const tokenMatch = limit.token === "*" || limit.token === extractToken(ctx);
    if (!chainMatch || !tokenMatch) continue;

    for (const period of ["daily", "weekly", "monthly"] as const) {
      const cap = limit[period];
      if (!cap) continue;

      const spent = getSpentInPeriod(ledger, ctx.apiKeyId, limit.chainId, limit.token, period);
      const capNum = parseFloat(cap);

      if (spent + txValue > capNum) {
        const result: PolicyResult = {
          allow: false,
          reason: `${period.charAt(0).toUpperCase() + period.slice(1)} ${limit.token === "*" ? "" : limit.token + " "}limit on ${limit.chainId === "*" ? "all chains" : limit.chainId} exceeded: $${(spent + txValue).toFixed(2)}/$${capNum.toFixed(2)}`,
        };
        logResult(ctx, "aegis-budget", result);
        process.stdout.write(JSON.stringify(result));
        return;
      }
    }
  }

  const result: PolicyResult = { allow: true };
  logResult(ctx, "aegis-budget", result);
  process.stdout.write(JSON.stringify(result));
}

function estimateValue(ctx: PolicyContext): number {
  // Use simulation result if available
  if (ctx.simulation?.value) {
    return parseFloat(ctx.simulation.value);
  }
  // Fall back to tx value field
  if (ctx.transaction.value) {
    return parseFloat(ctx.transaction.value);
  }
  return 0;
}

function extractToken(ctx: PolicyContext): string {
  // Check if tx data looks like an ERC-20 transfer (0xa9059cbb)
  if (ctx.transaction.data?.startsWith("0xa9059cbb")) {
    return "ERC20"; // Simplified; real impl would decode token address
  }
  // Default to native token based on chain
  if (ctx.chainId.startsWith("eip155:")) return "ETH";
  if (ctx.chainId.startsWith("solana:")) return "SOL";
  return "UNKNOWN";
}

function logResult(ctx: PolicyContext, policyName: string, result: PolicyResult): void {
  appendPolicyLog({
    timestamp: new Date().toISOString(),
    policyName,
    apiKeyId: ctx.apiKeyId,
    chainId: ctx.chainId,
    allowed: result.allow,
    reason: result.reason,
    transactionPreview: ctx.transaction.to
      ? `${ctx.transaction.to?.slice(0, 10)}...`
      : undefined,
  });
}

main().catch((err) => {
  const result: PolicyResult = {
    allow: false,
    reason: `aegis-budget error: ${(err as Error).message}`,
  };
  process.stdout.write(JSON.stringify(result));
  process.exit(1);
});
```

// packages/policies/src/guard/index.ts
#!/usr/bin/env node

import {
  type PolicyContext,
  type PolicyResult,
  readGuardConfig,
  appendPolicyLog,
} from "@aegis-ows/shared";

async function main(): Promise<void> {
  const chunks: Buffer[] = [];
  for await (const chunk of process.stdin) {
    chunks.push(chunk as Buffer);
  }
  const ctx: PolicyContext = JSON.parse(Buffer.concat(chunks).toString("utf-8"));

  const config = readGuardConfig();
  const targetAddress = extractTargetAddress(ctx);

  if (!targetAddress) {
    const result: PolicyResult = { allow: true, reason: "No target address to check" };
    logResult(ctx, "aegis-guard", result);
    process.stdout.write(JSON.stringify(result));
    return;
  }

  // Always block known bad addresses
  if (config.blockAddresses?.some((a) => a.toLowerCase() === targetAddress.toLowerCase())) {
    const result: PolicyResult = {
      allow: false,
      reason: `Address ${targetAddress.slice(0, 10)}... is on the blocklist`,
    };
    logResult(ctx, "aegis-guard", result);
    process.stdout.write(JSON.stringify(result));
    return;
  }

  // Allowlist mode: block if not explicitly allowed
  if (config.mode === "allowlist") {
    const allowed = getAllowedAddresses(config.addresses, ctx.chainId);
    if (allowed.length > 0 && !allowed.some((a) => a.toLowerCase() === targetAddress.toLowerCase())) {
      const result: PolicyResult = {
        allow: false,
        reason: `Address ${targetAddress.slice(0, 10)}... not in allowlist for ${ctx.chainId}`,
      };
      logResult(ctx, "aegis-guard", result);
      process.stdout.write(JSON.stringify(result));
      return;
    }
  }

  const result: PolicyResult = { allow: true };
  logResult(ctx, "aegis-guard", result);
  process.stdout.write(JSON.stringify(result));
}

function extractTargetAddress(ctx: PolicyContext): string | null {
  return ctx.transaction.to ?? null;
}

function getAllowedAddresses(addresses: Record<string, string[]>, chainId: string): string[] {
  const exact = addresses[chainId] ?? [];
  // Check wildcard entries (e.g., "solana:*")
  const wildcardKey = Object.keys(addresses).find((k) => {
    if (!k.includes("*")) return false;
    const prefix = k.replace("*", "");
    return chainId.startsWith(prefix);
  });
  const wildcard = wildcardKey ? addresses[wildcardKey] : [];
  return [...exact, ...wildcard];
}

function logResult(ctx: PolicyContext, policyName: string, result: PolicyResult): void {
  appendPolicyLog({
    timestamp: new Date().toISOString(),
    policyName,
    apiKeyId: ctx.apiKeyId,
    chainId: ctx.chainId,
    allowed: result.allow,
    reason: result.reason,
    transactionPreview: ctx.transaction.to
      ? `${ctx.transaction.to.slice(0, 10)}...`
      : undefined,
  });
}

main().catch((err) => {
  const result: PolicyResult = {
    allow: false,
    reason: `aegis-guard error: ${(err as Error).message}`,
  };
  process.stdout.write(JSON.stringify(result));
  process.exit(1);
});

// packages/policies/src/approve/index.ts
#!/usr/bin/env node

import {
  type PolicyContext,
  type PolicyResult,
  readApproveConfig,
  readApprovals,
  addPendingApproval,
  isApproved,
  appendPolicyLog,
} from "@aegis-ows/shared";

async function main(): Promise<void> {
  const chunks: Buffer[] = [];
  for await (const chunk of process.stdin) {
    chunks.push(chunk as Buffer);
  }
  const ctx: PolicyContext = JSON.parse(Buffer.concat(chunks).toString("utf-8"));

  const config = readApproveConfig();
  const txValue = estimateValue(ctx);

  const autoApprove = parseFloat(config.thresholds.auto_approve_below);
  const requireApproval = parseFloat(config.thresholds.require_approval_above);
  const hardBlock = parseFloat(config.thresholds.hard_block_above);

  // Hard block
  if (txValue > hardBlock) {
    const result: PolicyResult = {
      allow: false,
      reason: `Transaction value $${txValue.toFixed(2)} exceeds hard limit of $${hardBlock.toFixed(2)}`,
    };
    logResult(ctx, "aegis-approve", result);
    process.stdout.write(JSON.stringify(result));
    return;
  }

  // Auto approve
  if (txValue <= autoApprove) {
    const result: PolicyResult = { allow: true };
    logResult(ctx, "aegis-approve", result);
    process.stdout.write(JSON.stringify(result));
    return;
  }

  // Check if already approved
  if (isApproved(ctx.apiKeyId, ctx.chainId, ctx.transaction.to)) {
    const result: PolicyResult = { allow: true, reason: "Previously approved" };
    logResult(ctx, "aegis-approve", result);
    process.stdout.write(JSON.stringify(result));
    return;
  }

  // Requires approval — add to queue
  const pending = addPendingApproval({
    apiKeyId: ctx.apiKeyId,
    chainId: ctx.chainId,
    transaction: ctx.transaction,
    estimatedValue: txValue.toFixed(2),
    token: extractToken(ctx),
    reason: `Transaction value $${txValue.toFixed(2)} requires approval (threshold: $${requireApproval.toFixed(2)})`,
    ttlMinutes: config.approval_ttl_minutes,
  });

  const result: PolicyResult = {
    allow: false,
    reason: `Awaiting human approval (${pending.id}). Review at dashboard or run: aegis approve ${pending.id}`,
  };
  logResult(ctx, "aegis-approve", result);
  process.stdout.write(JSON.stringify(result));
}

function estimateValue(ctx: PolicyContext): number {
  if (ctx.simulation?.value) return parseFloat(ctx.simulation.value);
  if (ctx.transaction.value) return parseFloat(ctx.transaction.value);
  return 0;
}

function extractToken(ctx: PolicyContext): string {
  if (ctx.transaction.data?.startsWith("0xa9059cbb")) return "ERC20";
  if (ctx.chainId.startsWith("eip155:")) return "ETH";
  if (ctx.chainId.startsWith("solana:")) return "SOL";
  return "UNKNOWN";
}

function logResult(ctx: PolicyContext, policyName: string, result: PolicyResult): void {
  appendPolicyLog({
    timestamp: new Date().toISOString(),
    policyName,
    apiKeyId: ctx.apiKeyId,
    chainId: ctx.chainId,
    allowed: result.allow,
    reason: result.reason,
    transactionPreview: ctx.transaction.to
      ? `${ctx.transaction.to.slice(0, 10)}...`
      : undefined,
  });
}

main().catch((err) => {
  const result: PolicyResult = {
    allow: false,
    reason: `aegis-approve error: ${(err as Error).message}`,
  };
  process.stdout.write(JSON.stringify(result));
  process.exit(1);
});

      console.log(`Removed ${opts.remove} from all lists`);
      return;
    }

    // Show current config
    console.log(`Guard Mode: ${config.mode}\n`);
    console.log("Allowlist:");
    for (const [chain, addrs] of Object.entries(config.addresses)) {

import { Command } from "commander";
import { readLedger } from "@aegis-ows/shared";

export const reportCommand = new Command("report")
  .description("Generate spending report")
  .option("--period <period>", "today, week, month, all", "today")
  .option("--format <format>", "summary, detailed, csv", "summary")
  .action((opts: { period: string; format: string }) => {

    const now = new Date();
    let cutoff: Date;

    switch (opts.period) {
      case "today":

        break;
      case "week": {
        const day = now.getDay();
        cutoff = new Date(now.getFullYear(), now.getMonth(), now.getDate() - day);
        break;

// packages/mcp-server/src/tools/check-budget.ts
import { z } from "zod";
import { readBudgetConfig, readLedger, getSpentInPeriod } from "@aegis-ows/shared";

export const checkBudgetSchema = {
  walletName: z.string().describe("OWS wallet name"),
  chain: z.string().optional().describe("Optional: check specific chain (CAIP-2)"),
  period: z.enum(["daily", "weekly", "monthly"]).optional().describe("Budget period to check"),
};

export async function checkBudget(params: {
  walletName: string;
  chain?: string;
  period?: "daily" | "weekly" | "monthly";
}) {
  const config = readBudgetConfig();
  const ledger = readLedger();
  const periods = params.period ? [params.period] : (["daily", "weekly", "monthly"] as const);

  const budgets: Array<{
    chain: string;
    token: string;
    period: string;
    spent: string;
    limit: string;
    remaining: string;
  }> = [];

  for (const limit of config.limits) {
    if (params.chain && limit.chainId !== params.chain && limit.chainId !== "*") continue;

    for (const period of periods) {
      const cap = limit[period];
      if (!cap) continue;

      const spent = getSpentInPeriod(ledger, "*", limit.chainId, limit.token, period);
      const capNum = parseFloat(cap);

      budgets.push({
        chain: limit.chainId,
        token: limit.token,
        period,
        spent: spent.toFixed(2),
        limit: capNum.toFixed(2),
        remaining: (capNum - spent).toFixed(2),
      });
    }
  }

  return {
    wallet: params.walletName,
    budgets,
  };
}

import { appendLedgerEntry } from "@aegis-ows/shared";
import type { LedgerEntry } from "@aegis-ows/shared";

export interface X402PaymentDetails {
  network: string;
  token: string;
  amount: string;
  recipient: string;
  resource: string;
}

export function parsePaymentRequired(headers: Headers): X402PaymentDetails | null {
  const paymentHeader = headers.get("x-payment") || headers.get("payment-required");
  if (!paymentHeader) return null;
  try {
    return JSON.parse(paymentHeader) as X402PaymentDetails;
  } catch {
    return null;
  }
}

export async function payX402(params: {
  url: string;
  walletName: string;
  chain?: string;
  maxAmount?: string;
}): Promise<{ success: boolean; content?: string; error?: string; payment?: X402PaymentDetails }> {
  // Step 1: Fetch the URL to get the 402 response
  const initialResponse = await fetch(params.url);

  if (initialResponse.status !== 402) {
    // Not a paid resource, just return the content
    const content = await initialResponse.text();
    return { success: true, content };
  }

  // Step 2: Parse payment requirements from 402 response
  const payment = parsePaymentRequired(initialResponse.headers);
  if (!payment) {
    return { success: false, error: "402 response missing payment details in headers" };
  }

  // Step 3: Check maxAmount constraint
  if (params.maxAmount && parseFloat(payment.amount) > parseFloat(params.maxAmount)) {
    return {
      success: false,
      error: `Price $${payment.amount} exceeds max amount $${params.maxAmount}`,
      payment,
    };
  }

  // Step 4: Sign payment via OWS
  // In a real implementation, this would call:
  //   const ows = require("@open-wallet-standard/core");
  //   const signature = await ows.signTransaction(walletName, chain, paymentTx);
  // For hackathon demo, we simulate the signing
  const mockSignature = `aegis-sig-${Date.now()}`;

  // Step 5: Retry with payment signature
  const paidResponse = await fetch(params.url, {
    headers: {
      "x-payment-signature": mockSignature,
      "x-payment-network": payment.network,
      "x-payment-token": payment.token,
      "x-payment-amount": payment.amount,
    },
  });

  if (!paidResponse.ok) {
    return { success: false, error: `Payment verification failed: ${paidResponse.status}`, payment };
  }

  const content = await paidResponse.text();

  // Step 6: Log to ledger
  const entry: LedgerEntry = {
    timestamp: new Date().toISOString(),
    apiKeyId: params.walletName,
    chainId: payment.network,
    token: payment.token,
    amount: payment.amount,
    tool: "aegis_pay_x402",
    description: `x402 payment to ${params.url}`,
  };
  appendLedgerEntry(entry);

  return { success: true, content, payment };
}

import { appendLedgerEntry } from "@aegis-ows/shared";
import type { LedgerEntry } from "@aegis-ows/shared";
import { readFileSync, writeFileSync, existsSync } from "node:fs";
import { PATHS, ensureAegisDir } from "@aegis-ows/shared";

interface MppSession {
  sessionId: string;
  walletName: string;
  endpoint: string;
  maxCost: string;
  spent: string;
  createdAt: string;
}

interface MppSessions {
  sessions: MppSession[];
}

function readSessions(): MppSessions {
  if (!existsSync(PATHS.mppSessions)) return { sessions: [] };
  return JSON.parse(readFileSync(PATHS.mppSessions, "utf-8")) as MppSessions;
}

function writeSessions(data: MppSessions): void {
  ensureAegisDir();
  writeFileSync(PATHS.mppSessions, JSON.stringify(data, null, 2));
}

export async function payMpp(params: {
  url: string;
  walletName: string;
  sessionId?: string;
  maxSessionCost?: string;
}): Promise<{ success: boolean; content?: string; error?: string; sessionId?: string }> {
  // Step 1: Fetch to check for 402
  const initial = await fetch(params.url);

  if (initial.status !== 402) {
    return { success: true, content: await initial.text() };
  }

  // Step 2: Parse MPP challenge
  const challenge = await initial.json().catch(() => null) as { challengeId?: string } | null;
  if (!challenge?.challengeId) {
    return { success: false, error: "402 response missing MPP challenge" };
  }

  // Step 3: Get or create session
  const sessions = readSessions();
  let session: MppSession;

  if (params.sessionId) {
    const existing = sessions.sessions.find((s) => s.sessionId === params.sessionId);
    if (!existing) return { success: false, error: `Session ${params.sessionId} not found` };
    session = existing;
  } else {
    session = {
      sessionId: `mpp-${Date.now()}`,
      walletName: params.walletName,
      endpoint: new URL(params.url).origin,
      maxCost: params.maxSessionCost ?? "10.00",
      spent: "0.00",
      createdAt: new Date().toISOString(),
    };
    sessions.sessions.push(session);
    writeSessions(sessions);
  }

  // Step 4: Authorize payment (mock OWS signing)
  const mockCredential = `aegis-mpp-${session.sessionId}-${Date.now()}`;

  // Step 5: Retry with authorization
  const paid = await fetch(params.url, {
    headers: { Authorization: `MPP ${mockCredential}` },
  });

  if (!paid.ok) {
    return { success: false, error: `MPP payment failed: ${paid.status}`, sessionId: session.sessionId };
  }

  const content = await paid.text();

  // Step 6: Log
  const entry: LedgerEntry = {
    timestamp: new Date().toISOString(),
    apiKeyId: params.walletName,
    chainId: "mpp",
    token: "USDC",
    amount: "0.01", // Simplified; real impl reads from receipt
    tool: "aegis_pay_mpp",
    description: `MPP payment to ${params.url} (session: ${session.sessionId})`,
  };
  appendLedgerEntry(entry);

  return { success: true, content, sessionId: session.sessionId };
}

// packages/mcp-server/src/tools/report.ts
import { z } from "zod";
import { readLedger } from "@aegis-ows/shared";

export const reportSchema = {
  walletName: z.string().describe("OWS wallet name"),
  period: z.enum(["today", "week", "month", "all"]).optional().describe("Report period"),
  format: z.enum(["summary", "detailed", "csv"]).optional().describe("Output format"),
};

export async function handleReport(params: {
  walletName: string;
  period?: string;
  format?: string;
}) {
  const ledger = readLedger();
  const now = new Date();
  let cutoff: Date;

  switch (params.period ?? "today") {
    case "today":
      cutoff = new Date(now.getFullYear(), now.getMonth(), now.getDate());
      break;
    case "week": {
      const day = now.getDay();
      cutoff = new Date(now.getFullYear(), now.getMonth(), now.getDate() - day);
      break;
    }
    case "month":
      cutoff = new Date(now.getFullYear(), now.getMonth(), 1);
      break;
    default:
      cutoff = new Date(0);
  }

  const entries = ledger.entries.filter((e) => new Date(e.timestamp) >= cutoff);
  const total = entries.reduce((s, e) => s + parseFloat(e.amount), 0);

  if (params.format === "csv") {
    const csv = ["timestamp,apiKeyId,chainId,token,amount,txHash,tool"]
      .concat(entries.map((e) => `${e.timestamp},${e.apiKeyId},${e.chainId},${e.token},${e.amount},${e.txHash ?? ""},${e.tool ?? ""}`))
      .join("\n");
    return { content: [{ type: "text" as const, text: csv }] };
  }

  if (params.format === "detailed") {
    const lines = entries.map(
      (e) => `${e.timestamp} | ${e.apiKeyId} | $${e.amount} ${e.token} | ${e.chainId} | ${e.tool ?? ""}`
    );
    return { content: [{ type: "text" as const, text: lines.join("\n") || "No transactions" }] };
  }

  // Summary
  const byChain: Record<string, number> = {};
  for (const e of entries) {
    byChain[e.chainId] = (byChain[e.chainId] ?? 0) + parseFloat(e.amount);
  }

  const chainLines = Object.entries(byChain)
    .map(([chain, amt]) => `  ${chain}: $${amt.toFixed(2)}`)
    .join("\n");

  return {
    content: [
      {
        type: "text" as const,
        text: `Spending Report (${params.period ?? "today"}) for ${params.walletName}:\n\nTotal: $${total.toFixed(2)} across ${entries.length} transactions\n\n${chainLines || "  No spending"}`,
      },
    ],
  };
}

// dashboard/src/lib/aegis-data.ts
import {
  readLedger,
  readPolicyLog,
  readApprovals,
  readRegistry,
  readBudgetConfig,
  readGuardConfig,
  readApproveConfig,
  getSpentInPeriod,
} from "@aegis-ows/shared";
import type { BudgetLedger, PolicyLog, ApprovalQueue, ServiceRegistry } from "@aegis-ows/shared";

export function getSpendingData() {
  const ledger = readLedger();
  const config = readBudgetConfig();

  const budgets = config.limits.map((limit) => {
    const periods = (["daily", "weekly", "monthly"] as const)
      .filter((p) => limit[p])
      .map((p) => ({
        period: p,
        spent: getSpentInPeriod(ledger, "*", limit.chainId, limit.token, p),
        limit: parseFloat(limit[p]!),
      }));

    return { chainId: limit.chainId, token: limit.token, periods };
  });

  const total = ledger.entries.reduce((s, e) => s + parseFloat(e.amount), 0);
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  const todayTotal = ledger.entries
    .filter((e) => new Date(e.timestamp) >= today)
    .reduce((s, e) => s + parseFloat(e.amount), 0);

  return { ledger, budgets, total, todayTotal };
}

export function getPolicyData() {
  const log = readPolicyLog();
  const budgetConfig = readBudgetConfig();
  const guardConfig = readGuardConfig();
  const approveConfig = readApproveConfig();
  return { log, budgetConfig, guardConfig, approveConfig };
}

export function getApprovalData() {
  return readApprovals();
}

export function getServiceData() {
  return readRegistry();
}

// dashboard/src/components/spending-chart.tsx
"use client";
import { BarChart, Bar, XAxis, YAxis, Tooltip, ResponsiveContainer, CartesianGrid } from "recharts";
import type { LedgerEntry } from "@aegis-ows/shared";

export function SpendingChart({ entries }: { entries: LedgerEntry[] }) {
  // Group by date
  const byDate: Record<string, number> = {};
  for (const e of entries) {
    const date = e.timestamp.slice(0, 10);
    byDate[date] = (byDate[date] ?? 0) + parseFloat(e.amount);
  }

  const data = Object.entries(byDate)
    .sort(([a], [b]) => a.localeCompare(b))
    .slice(-14)
    .map(([date, amount]) => ({ date: date.slice(5), amount: parseFloat(amount.toFixed(2)) }));

  if (data.length === 0) {
    return <p className="text-sm text-muted-foreground">No spending data yet.</p>;
  }

  return (
    <ResponsiveContainer width="100%" height={250}>
      <BarChart data={data}>
        <CartesianGrid strokeDasharray="3 3" className="stroke-muted" />
        <XAxis dataKey="date" className="text-xs" />
        <YAxis className="text-xs" tickFormatter={(v: number) => `$${v}`} />
        <Tooltip formatter={(v: number) => `$${v.toFixed(2)}`} />
        <Bar dataKey="amount" fill="hsl(var(--primary))" radius={[4, 4, 0, 0]} />
      </BarChart>
    </ResponsiveContainer>
  );
}

import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";

interface Approval {
  id: string;
  apiKeyId: string;
  chainId: string;
  transaction: { to?: string; value?: string };
  estimatedValue: string;
  token: string;
  reason?: string;
  createdAt: string;
  expiresAt: string;
  status: string;
}

export default function ApprovalsPage() {
  const [approvals, setApprovals] = useState<Approval[]>([]);

  const load = async () => {
    const res = await fetch("/api/approvals");
    const data = await res.json() as { pending: Approval[] };
    setApprovals(data.pending ?? []);

  useEffect(() => { load(); }, []);

  const handleAction = async (id: string, action: "approved" | "rejected") => {
    await fetch("/api/approvals", {
      method: "POST",

    });
    load();
  };

  const resolved = approvals.filter((a) => a.status !== "pending");

// demo/mock-x402-server.ts
import express from "express";

const app = express();
const PORT = 4020;

// Mock x402-protected endpoint
app.get("/api/scrape", (req, res) => {
  const sig = req.headers["x-payment-signature"];

  if (!sig) {
    // Return 402 with payment details
    res.status(402).setHeader(
      "x-payment",
      JSON.stringify({
        network: "eip155:1",
        token: "USDC",
        amount: "0.01",
        recipient: "0xSERVICE_WALLET",
        resource: "/api/scrape",
      })
    );
    res.json({ error: "Payment required", price: "0.01 USDC" });
    return;
  }

  // Payment provided — return content
  res.json({
    success: true,
    data: {
      title: "Example Scraped Page",
      content: "This is the scraped content that was paid for via x402.",
      timestamp: new Date().toISOString(),
    },
  });
});

app.listen(PORT, () => {
  console.log(`Mock x402 server running at http://localhost:${PORT}`);
  console.log("  GET /api/scrape — returns 402 without payment, content with payment");
});
